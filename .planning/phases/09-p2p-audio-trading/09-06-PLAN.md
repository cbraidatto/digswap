**Supabase admin mock pattern (createQueryChain):**
```typescript
function createQueryChain(finalReturn: { data: any; error: any }) {
  const chain: any = {};
  for (const method of ["from", "select", "insert", "update", "eq", "in", "single", "maybeSingle"]) {
    chain[method] = vi.fn().mockReturnValue(chain);
  }
  // Terminal methods return the final value
  chain.single = vi.fn().mockResolvedValue(finalReturn);
  chain.maybeSingle = vi.fn().mockResolvedValue(finalReturn);
  // For insert/update without .single()
  chain.select = vi.fn().mockReturnValue(chain);
  return chain;
}
```
