```typescript
export interface TransferCallbacks {
  onProgress(bytesTransferred: number, totalBytes: number): void;
  onComplete(filePath: string, sha256: string): void;
  onError(err: Error): void;
}

export const CHUNK_SIZE = 64 * 1024; // 64 KB

/** Sender: reads file from disk and sends in CHUNK_SIZE chunks. */
export async function sendFile(
  conn: DataConnection,
  filePath: string,
  callbacks: TransferCallbacks
): Promise<void> { ... }

/** Receiver: writes chunks to partPath, renames to finalPath on complete, verifies SHA-256. */
export async function receiveFile(
  conn: DataConnection,
  partPath: string,
  finalPath: string,
  expectedSha256: string | null,  // null = skip verification (sender side doesn't know yet)
  callbacks: TransferCallbacks
): Promise<void> { ... }
```

Protocol over DataChannel (sender → receiver):
1. Sender sends a JSON header message first:
   ```json
   { "type": "header", "filename": "...", "size": 123456, "sha256": "abc..." }
   ```
   SHA-256 is computed before sending starts (full file hash).
2. Sender sends raw ArrayBuffer chunks of CHUNK_SIZE.
3. Sender sends a final JSON message: `{ "type": "eof" }`

Receiver:
- Waits for `header` message, stores filename/size/sha256.
- Writes each ArrayBuffer chunk to `partPath` via `fs.appendFileSync` (or write stream).
- On `eof`: verifies SHA-256 of the written file; if match, renames `.part` to `finalPath`; calls `onComplete(finalPath, sha256)`.
- Calls `onProgress(bytesReceived, totalBytes)` after each chunk.

Backpressure (sender):
- After each chunk send, check `conn.dataChannel.bufferedAmount`. If above 1MB, pause and wait for `bufferedamountlow` event before continuing.

Cancel handling:
- Both sides: if `conn` closes mid-transfer, call `onError(new Error("Transfer cancelled"))`.
- Caller is responsible for deleting `.part` file on error.

```typescript
import { createHash } from "node:crypto";
import { createReadStream } from "node:fs";

async function computeFileSha256(filePath: string): Promise<string> {
  return new Promise((resolve, reject) => {
    const hash = createHash("sha256");
    createReadStream(filePath).on("data", (d) => hash.update(d)).on("end", () => resolve(hash.digest("hex"))).on("error", reject);
  });
}
```
