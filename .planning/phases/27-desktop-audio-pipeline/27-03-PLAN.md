    - MultiItemTransferCallbacks {
        onItemProgress(event: MultiItemProgressEvent): void;
        onItemComplete(index: number, proposalItemId: string, filePath: string, sha256: string): void;
        onBatchComplete(event: MultiItemCompleteEvent): void;
        onError(itemIndex: number, proposalItemId: string, err: Error): void;
      }

    sendMultiItemBatch(conn, tradeId, items: BatchTransferItem[], callbacks): Promise<void>
    - Test 1: Calls sendFile once per item for 1-item batch
    - Test 2: Calls sendFile 2 times sequentially for 2-item batch (second starts after first resolves)
    - Test 3: Calls sendFile 3 times sequentially for 3-item batch
    - Test 4: Fires onItemProgress with correct itemIndex and proposalItemId for each chunk
    - Test 5: Fires onItemComplete after each item's sendFile resolves
    - Test 6: Fires onBatchComplete with allVerified=true when all items complete
    - Test 7: If item 1 (index 0) fails, fires onError(0, ...) and continues to item 1 (index 1) — partial batch completes
    - Test 8: onBatchComplete has allVerified=false when any item errored

    receiveMultiItemBatch(conn, tradeId, items: Array<{proposalItemId, partPath, finalPath, expectedSha256}>, callbacks): Promise<void>
    - Test 9: Calls receiveFile once per item for 1-item batch
    - Test 10: Calls receiveFile 3 times sequentially for 3-item batch
    - Test 11: If item index 1 fails, resumes at index 1 on retry (caller re-invokes with startFromIndex=1)
    - Test 12: Fires onBatchComplete with completedItems containing only successful items

    Note: "resume from failed item" means the batch tracks a resumeFromIndex. On error,
    onError is called and the batch records the failure but continues to remaining items
    (non-blocking error model). The caller can re-invoke with startFromIndex to skip completed items.
    startFromIndex defaults to 0.
  </behavior>
