   ```typescript
   import { Tray, Menu, nativeImage, app } from "electron";
   import path from "node:path";
   import { focusMainWindow } from "./window";

   let tray: Tray | null = null;

   export function createTray(): void {
     // Try loading icon from resources, fallback to inline base64
     const iconPath = path.join(__dirname, "../../resources/tray-icon.png");
     let icon: Electron.NativeImage;
     try {
       icon = nativeImage.createFromPath(iconPath);
       if (icon.isEmpty()) throw new Error("empty");
     } catch {
       // Fallback: 16x16 white circle on transparent (minimal vinyl icon)
       icon = nativeImage.createFromDataURL(
         "data:image/png;base64,iVBORw0KGgoAAAANSUhEUgAAABAAAAAQCAYAAAAf8/9hAAAA..."
       );
     }

     tray = new Tray(icon);
     const contextMenu = Menu.buildFromTemplate([
       { label: "Open", click: () => focusMainWindow() },
       { type: "separator" },
       { label: "Quit", click: () => app.quit() },
     ]);
     tray.setToolTip("DigSwap Desktop");
     tray.setContextMenu(contextMenu);
     tray.on("double-click", () => focusMainWindow());
   }

   export function destroyTray(): void {
     tray?.destroy();
     tray = null;
   }
   ```
   - Store tray in module-level variable to prevent GC (per RESEARCH.md Pitfall 1)
   - Context menu has exactly 2 items: "Open" and "Quit" (per D-02)
   - Double-click opens/focuses window (per D-02)
